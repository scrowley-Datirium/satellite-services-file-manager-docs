import { Readable } from 'stream';

/**

 */
export interface ReadRequest {
    action: string;
    path: string;
    showHiddenItems: boolean;
    data: FileInformation[];
    token: string;
    ftp: boolean;
    names?: string[];
    ftpInformation?: FTPInformation;
    newName?: string;
    name?: string;
    target?: string;
    currentLocation?: string;
}

/** interface representing options for ftp usage */
export interface FTPInformation {
    /** url/ip for ftp */
    address: string;
    /** username for ftp */
    username: string;
    /** password for ftp */
    password: string;
}

export interface Aria2cResponse {
    guid: string;
    location: string;
}

/** interface representing a files information (not content) */
export interface FileInformation {
    /** name of file */
    name: string;
    /** date file was created @type {Date} */
    dateCreated: Date;
    /** date file was last modified @type {Date} */
    dateModified: Date;
    filterPath: string;
    hasChild: boolean;
    /** if file is a physical file with contents or not */
    isFile: boolean;
    /** size of file */
    size: string | number;
    type: string;
    sample_id?: string;
    target?: string;
}

/** interface represnting an error thrown by file-manager */
export interface FileManagerError {
    code: string;
    /** error message */
    message: string;
    fileExists: string[];
}

/**
 * response from getting information from a file (or folder)
 */
export interface ReadResponse {
    /** information about file/folder */
    cwd: FileInformation;
    /** if folder, include list of information for nested files in that folder */
    files: FileInformation[];
    /** error from trying to get a files info */
    error?: FileManagerError;
}

/** @deprecated (not in use?) */
export interface DownloadRequestFromFM {
    downloadInput: string;
}

/**
 * @deprecated
 * (not in use?)
 */
export interface DownloadInputFromFM {
    action: string;
    path: string;
    names: string[];
    data: FileInformation[];
    token: string;
}

/**
 * interface representing the api-response from sending a satellite a 'download' request
 */
export interface DownloadToSatelliteRequest {
    /** information about file that was downloaded */
    details: FileInformation[];
    /** path to file on satellite */
    path: string;
    /** satellite token */
    token: string;
}

/** interface representing part of the reponse from file-managers 'fileOperations' endpoint (used when request.action is 'details') */
export interface FileDetails {
    /** name of file */
    name: string;
    /** size of file */
    size: string;
    /** when file was created */
    created: Date;
    /** when file was modified */
    modified: Date;
    /** location of file on satellite */
    location: string;
    /** if file is FILE (could be folder) */
    isFile: boolean;
    type: string;
    multipleFiles?: boolean;
}


/** interface representing response from fs.Promises stat() method */
export interface fsStats {
    /** size of file */
    size: number;
    ctime: string;
    mtime: string;
    isFile(): boolean;
    isDirectory(): boolean;
}

/**
 * response from download request
 */
export interface DownloadResponse {
    stream: Readable;
    /** name of file */
    filename: string;
    /** size of file (bytes?) */
    filesize: number;
}

export interface DetailsResponse {
    details?: FileDetails;
}

/** interface representing a progress report on how an ftp download/upload is going */
export interface FTPProgress {
    name: string;
    type: string;
    /** how many bytes have been transferred so far */
    bytes: number;
    /** how many bytes are to be transferred overall */
    bytesOverall: number;
}

/** interface representing a request that a file be uploaded (satellite GAINS a file) */
export interface UploadRequest {
    /** url/path for upload */
    path: string;
    action: string;
    data: string;
    target: string;
    /** token for authorizing the upload request */
    token: string;
}

/** interface representing information from an upload */
export interface UploadInformation {
    fieldname: string;
    originalname: string;
    encoding: string;
    mimetype: string;
    buffer: Buffer;
    size: number;
    /** location of file */
    destination: string;
    /** tmp name of file */
    filename: string; 
    /** destination + filename */
    path: string; 
}

export interface RootLocations {
    projects: string;
    labs: string;
    local: string;
}

/** interface representing token information for file-manager after being decoded */
export interface FileManagerToken {
    /** project id correlating to file related request */
    projectId: string;
    /** sample id correlating to file related request */
    sampleId?: string;
    /** satellite id correlating to file related request */
    satelliteId: string;
    iat: number;
    exp: number;
}
